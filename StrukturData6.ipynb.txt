        "def insertion_sort(arr):\n",
        "  n = len(arr)\n",
        "  for i in range(1, n):\n",
        "    key = arr[i]\n",
        "    j = i - 1\n",
        "    while j >= 0 and arr[j] > key:\n",
        "      arr[j + 1] = arr[j]\n",
        "      j -= 1\n",
        "    arr[j + 1] = key\n",
        "  return arr\n",

        "  if len(arr) <= 1:\n",
        "    return arr\n",

        "  middle = [x for x in arr if x == pivot]\n",
        "  right = [x for x in arr if x >pivot]\n",
        "  return quick_sort(left) + middle + quick_sort(right)\n",
